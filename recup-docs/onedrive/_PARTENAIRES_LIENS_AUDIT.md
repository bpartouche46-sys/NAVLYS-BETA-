## ANNEXE A — Composant React `<AffiliateRedirect>` (à coller dans `navlys/components/`)

```tsx
// components/AffiliateRedirect.tsx
"use client";
import { useEffect } from "react";

type Props = {
  href: string;          // URL CPA réelle (ex: traderepublic.com/r/XYZ)
  program: string;       // slug interne (ex: "trade-republic")
  children: React.ReactNode;
};

export default function AffiliateRedirect({ href, program, children }: Props) {
  const onClick = (e: React.MouseEvent) => {
    e.preventDefault();
    // Cookie 30j tracking interne
    document.cookie = `aff_${program}=1; max-age=2592000; path=/; samesite=lax`;
    // Ping analytics (sans PII)
    if (typeof window !== "undefined" && "fetch" in window) {
      fetch("/api/aff-click", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ program, t: Date.now() })
      }).catch(() => {});
    }
    // Redirect
    window.location.href = href;
  };
  return (
    <a href={href} onClick={onClick} target="_blank" rel="sponsored noopener noreferrer">
      {children}
    </a>
  );
}
```

> `rel="sponsored"` = obligatoire Google guidelines + conformité FTC/AMF.

## ANNEXE B — Script de remplacement HTML (placeholder → CPA réel)

```bash
#!/usr/bin/env bash
# replace-affiliate-placeholders.sh
# Usage: ./replace-affiliate-placeholders.sh _SITES_MASTER/_BRUNOPARTOUCHE_AVEC_PARTENAIRES.html

set -euo pipefail
F="$1"
cp "$F" "$F.bak.$(date +%Y%m%d-%H%M)"

# Mapping placeholder → CPA réel (à compléter au fur et à mesure des inscriptions)
declare -A MAP=(
  ["traderepublic.com/?ref=BP001"]="traderepublic.com/r/REPLACE_ME"
  ["binance.com/?ref=BP001"]="binance.com/join?ref=REPLACE_ME"
  ["alpaca.markets/?ref=BP001"]="alpaca.markets/?afid=REPLACE_ME"
  ["etoro.com/?ref=BP001"]="etoro.com/B/REPLACE_ME"
  ["kraken.com/?ref=BP001"]="kraken.com/?clickref=REPLACE_ME"
  ["coinbase.com/?ref=BP001"]="coinbase.com/join/REPLACE_ME"
  ["tradingview.com/?ref=BP001"]="tradingview.com/?aff_id=REPLACE_ME"
  ["notion.so/?ref=BP001"]="affiliate.notion.so/REPLACE_ME"
  ["revolut.com/?ref=BP001"]="revolut.com/referral/REPLACE_ME"
  ["wise.com/?ref=BP001"]="wise.com/invite/REPLACE_ME"
  ["boursedirect.fr/?ref=BP001"]="trk.effiliation.com/redirect/REPLACE_ME"
  ["boursobank.com"]="trk.effiliation.com/redirect/REPLACE_ME"
  ["bybit.com/?ref=BP001"]="partner.bybit.com/b/REPLACE_ME"
  ["okx.com/?ref=BP001"]="okx.com/join/REPLACE_ME"
)

for K in "${!MAP[@]}"; do
  V="${MAP[$K]}"
  if grep -q "$K" "$F"; then
    sed -i.tmp "s|$K|$V|g" "$F" && rm -f "$F.tmp"
    echo "✅ remplacé: $K → $V"
  else
    echo "—  pas trouvé: $K"
  fi
done
echo "Done. Backup: $F.bak.*"
```

> Lancer sur chaque `*.html` partenaire. Remplacer `REPLACE_ME` au fur et à mesure des inscriptions affilié (manuel — un seul `sed -i` final par partenaire).
